🗄️ Supabase Database Setup
Run the SQL below once in your Supabase project → SQL Editor. This creates the tables and inserts the seed rows that all three pages (Order, Kitchen, Admin) depend on.
Step 1 — Create Tables + Seed Rows
-- ① settings table (menu_data + store_status)
create table if not exists public.settings (
  key        text primary key,
  value      text,
  out_of_stock text default '[]'
);

-- Seed store_status (open by default, no items out of stock)
insert into public.settings (key, value, out_of_stock)
values ('store_status', 'open', '[]')
on conflict (key) do nothing;

-- Seed menu_data with an empty placeholder
-- (the Admin page will overwrite this with real menu data on first save)
insert into public.settings (key, value)
values ('menu_data', '{"version":1,"categories":[],"items":[]}')
on conflict (key) do nothing;

-- ② orders table
create table if not exists public.orders (
  id           bigserial primary key,
  order_id     text unique not null,
  table_no     text,
  phone_no     text,
  customer_name  text,
  customer_email text,
  user_id      text,
  items        jsonb,
  total        numeric,
  note         text,
  status       text default 'pending',
  created_at   timestamptz default now()
);

-- ③ user_favourites table
create table if not exists public.user_favourites (
  user_id   text primary key,
  item_ids  integer[] default '{}'
);

-- ④ Enable Row Level Security and allow all reads/writes
-- (tighten these for production with proper policies)
alter table public.settings        enable row level security;
alter table public.orders          enable row level security;
alter table public.user_favourites enable row level security;

-- Allow full access (public) — refine for production
create policy "allow_all_settings"        on public.settings        for all using (true) with check (true);
create policy "allow_all_orders"          on public.orders          for all using (true) with check (true);
create policy "allow_all_user_favourites" on public.user_favourites for all using (true) with check (true);

-- ⑤ Enable real-time replication on all tables
alter publication supabase_realtime add table public.settings;
alter publication supabase_realtime add table public.orders;
Step 2 — Get your Supabase credentials
  1. Go to app.supabase.com → your project → Settings → API
  2. Copy Project URL and anon / public key
  3. Paste both into the SUPABASE_URL and SUPABASE_ANON_KEY constants at the top of all three files
Step 3 — Test the connection
⚡ How it all connects
• Admin page → writes settings.menu_data (JSON blob with all categories + items)
• Order page → reads settings.menu_data on load; subscribes for live updates. Writes to orders table on each order.
• Kitchen page → reads today's orders; subscribes for new ones. Updates settings.store_status (open/closed + out_of_stock list). Reads menu for Edit Order dropdown.
• Out-of-stock toggled in Kitchen → instantly reflected on Order page via real-time subscription to settings.store_status
• Menu changes in Admin → both Order and Kitchen pages update live within seconds
